% --- 10 TEST SCENARIOS FOR COMPARATIVE CUCKER-SMALE SIMULATION ---
% Each command can be copy-pasted directly into the MATLAB Command Window.

% Scenario 1: Standard Balanced Flocking (2 groups)
x1 = [5+rand(1,5), 15+rand(1,5)]; y1 = 10+rand(1,10); z1 = 10+rand(1,10); vx1 = [ones(1,5), -ones(1,5)]; vy1 = zeros(1,10); vz1 = zeros(1,10); tau1 = 0.05*rand(10,10);
run_simulation(x1, y1, z1, vx1, vy1, vz1, 0.5, tau1, 0.01, 0.3, 0.4, 0.1)

% Scenario 2: High Sensitivity (High Alpha, Low Beta)
x2 = rand(1,10)*20; y2 = rand(1,10)*20; z2 = rand(1,10)*20; vx2 = randn(1,10); vy2 = randn(1,10); vz2 = randn(1,10); tau2 = 0.02*rand(10,10);
run_simulation(x2, y2, z2, vx2, vy2, vz2, 0.4, tau2, 0.005, 0.5, 0.1, 0.05)

% Scenario 3: Large Propagation Delay effect (High Tau Factor)
x3 = [0:2:18]; y3 = zeros(1,10); z3 = zeros(1,10); vx3 = ones(1,10); vy3 = randn(1,10)*0.1; vz3 = randn(1,10)*0.1; tau3 = zeros(10,10);
run_simulation(x3, y3, z3, vx3, vy3, vz3, 0.5, tau3, 0.05, 0.2, 0.5, 0.1)

% Scenario 4: Fast Dynamics / Small Step (h=0.1)
x4 = rand(1,8)*5; y4 = rand(1,8)*5; z4 = rand(1,8)*5; vx4 = randn(1,8)*2; vy4 = randn(1,8)*2; vz4 = randn(1,8)*2; tau4 = 0.01*rand(8,8);
run_simulation(x4, y4, z4, vx4, vy4, vz4, 0.1, tau4, 0.002, 0.4, 0.4, 0.05)

% Scenario 5: Distant Clusters merging
x5 = [zeros(1,5), ones(1,5)*50]; y5 = rand(1,10)*10; z5 = rand(1,10)*10; vx5 = [ones(1,5)*2, -ones(1,5)*2]; vy5 = zeros(1,10); vz5 = zeros(1,10); tau5 = 0.05*eye(10);
run_simulation(x5, y5, z5, vx5, vy5, vz5, 0.5, tau5, 0.01, 0.3, 0.3, 0.1)

% Scenario 6: Random Cloud (Stress Test)
x6 = rand(1,15)*30; y6 = rand(1,15)*30; z6 = rand(1,15)*30; vx6 = randn(1,15); vy6 = randn(1,15); vz6 = randn(1,15); tau6 = 0.1*rand(15,15);
run_simulation(x6, y6, z6, vx6, vy6, vz6, 0.5, tau6, 0.02, 0.25, 0.4, 0.1)

% Scenario 7: Heavy Decay (High Beta = 1.0)
x7 = rand(1,10)*10; y7 = rand(1,10)*10; z7 = rand(1,10)*10; vx7 = randn(1,10); vy7 = randn(1,10); vz7 = randn(1,10); tau7 = 0.03*rand(10,10);
run_simulation(x7, y7, z7, vx7, vy7, vz7, 0.5, tau7, 0.01, 0.4, 1.0, 0.1)

% Scenario 8: Near Consensus (Tight Convergence)
x8 = rand(1,6)*5; y8 = rand(1,6)*5; z8 = rand(1,6)*5; vx8 = [1.0, 1.1, 0.9, 1.05, 0.95, 1.0]; vy8 = zeros(1,6); vz8 = zeros(1,6); tau8 = 0.1*rand(6,6);
run_simulation(x8, y8, z8, vx8, vy8, vz8, 0.5, tau8, 0.01, 0.3, 0.4, 0.001)

% Scenario 9: Opposing Vectors (Symmetry break)
x9 = [10, 10]; y9 = [0, 20]; z9 = [0, 0]; vx9 = [0, 0]; vy9 = [5, -5]; vz9 = [0, 0]; tau9 = [0 0.5; 0.5 0];
run_simulation(x9, y9, z9, vx9, vy9, vz9, 0.2, tau9, 0.05, 0.5, 0.2, 0.1)

% Scenario 10: Mixed Chaos
x10 = rand(1,12)*40; y10 = rand(1,12)*40; z10 = rand(1,12)*40; vx10 = randn(1,12)*5; vy10 = randn(1,12)*5; vz10 = randn(1,12)*5; tau10 = 0.05*rand(12,12);
run_simulation(x10, y10, z10, vx10, vy10, vz10, 0.5, tau10, 0.015, 0.35, 0.45, 0.1)
